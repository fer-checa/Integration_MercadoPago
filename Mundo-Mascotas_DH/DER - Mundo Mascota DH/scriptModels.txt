-- TABLAS

CREATE TABLE `users` (
   `id` INT AUTO_INCREMENT,
   `name` VARCHAR(255) NOT NULL,
   `email` VARCHAR(255) NOT NULL,
   `password` VARCHAR(255) NOT NULL,
   `image` VARCHAR(255) NOT NULL,
   `role_id` INT,
   PRIMARY KEY (`id`)
);

CREATE TABLE `products` (
   `id` INT AUTO_INCREMENT,
   `name` VARCHAR(255) NOT NULL,
   `description` VARCHAR(1000),
   `image` VARCHAR(255),
   `categoryAnimal_id` INT NOT NULL,
   `familyProduct_id` INT NOT NULL,
   `price` DOUBLE(8, 2),
   `discount` DOUBLE(8, 2),
   `date_release` DATETIME,
   `user_id` INT,
   `active` BIT NOT NULL DEFAULT true,
   PRIMARY KEY (`id`)
);

CREATE TABLE `familyProducts` (
   `id` INT AUTO_INCREMENT,
   `name` VARCHAR(255) NOT NULL,
   `user_id` INT,
   `active` BIT NOT NULL DEFAULT true,
   `date_release` DATETIME,
   PRIMARY KEY (`id`)
);

CREATE TABLE `categoryAnimals` (
   `id` INT AUTO_INCREMENT,
   `name` VARCHAR(255) NOT NULL,
   `user_id` INT,
   `active` BIT DEFAULT true,
   `date_release` DATETIME,
   PRIMARY KEY (`id`)
);

CREATE TABLE `orders` (
   `id` INT AUTO_INCREMENT,
   `user_id` INT,
   `date_init` DATETIME,
   `date_confirm` DATETIME,
   PRIMARY KEY (`id`)
);

CREATE TABLE `ordersDetails` (
   `id` INT AUTO_INCREMENT,
   `count` SMALLINT,
   `price_unit` DOUBLE(8, 2),
   `discount` DOUBLE(8, 2),
   PRIMARY KEY (`id`)
);

CREATE TABLE `productsOrdersDetails` (
   `id` INT,
   `order_id` INT,
   `order_detail_id` INT,
   `product_id` INT,
   PRIMARY KEY (`id`)
);

CREATE TABLE `roles` (
   `id` INT AUTO_INCREMENT,
   `role` VARCHAR(50) NOT NULL,
   PRIMARY KEY (`id`)
);
